\begin{frame}[fragile]
  \frametitle{Unprescribed static bounds in Python}

  \begin{lstlisting}[language=Python]
  def foo(x):
      if isinstance(x, int):
          return x + 1 
      elif isinstance(x, str): 
          return x + "abc"
  \end{lstlisting}
\end{frame}

\begin{frame}[fragile]
  \frametitle{Prescribed static bounds with datatypes}

  \begin{lstlisting}[language=ML]
  datatype int_or_str = 
      Int of int | 
      Str of string

  fun foo(Int x) = x + 1
    | foo(Str x) = x ^ "abc"
  \end{lstlisting}

\end{frame}
